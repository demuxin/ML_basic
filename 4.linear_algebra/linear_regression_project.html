def matxMultiply(A, B):
    M = [[] for i in range(shape(A)[0])]
    if shape(A)[1] != shape(B)[0]: # AB
        raise ValueError

    for a in range(shape(A)[0]): # A
        for b in range(shape(B)[1]): # B
            br = [B[i][b] for i in range(shape(B)[0])] # 
            M[a].append(sum([x*y for x, y in zip(A[a], br)]))

    return M

""" Gaussian Jordan  Ax = b.
    
        A:  
        b: 
        decPts: 4
        epsilon: 0 1.0e-16

     x  Ax = b
    None Ab 
    None A 
"""
def gj_Solve(A, b, decPts=4, epsilon=1.0e-16):
    if shape(A)[0] != shape(b)[0]:
        return False

    Ab = augmentMatrix(A, b) # Ab
    for i in range(shape(A)[1]): # 
        Abc = [Ab[j][i] for j in range(shape(Ab)[0])] # 

        Abc_max, Abc_max_index = 0, 0
        # 
        for index, x in enumerate(Abc[i:]):
            if Abc_max < abs(x):
                Abc_max = abs(x)
                max_index = index + i # 
                if x > 0:
                    sign = 1
                else:
                    sign = -1

        if Abc_max < epsilon:
            return None

# 
#         if Abc_max in Abc[i:]:
#             max_index = len(Abc) - Abc[::-1].index(Abc_max) - 1
#         else:
#             max_index = len(Abc) - Abc[::-1].index(-Abc_max) - 1

        # 1.(Abc)
        if max_index != i:
            swapRows(Ab, max_index, i)

        # 2.Abc1
        scaleRow(Ab, i, 1 / Abc_max * sign)

        Abc = [Ab[j][i] for j in range(shape(Ab)[0])] # 
        # 3.c0
        for index, x in enumerate(Abc):
            if abs(x) > epsilon and index != i:
                addScaledRow(Ab, index, i, -x)

    matxRound(Ab, decPts)

    # Ab
    result = [[] for _ in range(shape(Ab)[0])]
    for n in range(shape(Ab)[0]):
        result[n].append(Ab[n][-1])

    return result

def convert_2_dimen(X, Y):
    result_X = [[] for _ in range(len(X))]
